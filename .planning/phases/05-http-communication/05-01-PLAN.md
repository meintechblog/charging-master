    ```
      private eventBus: EventBus;
      private pollers: Map<string, NodeJS.Timeout> = new Map();
      private lastPersistedAt: Map<string, number> = new Map();

      // Same constants as MqttService:
      private readonly ACTIVE_POWER_THRESHOLD = 5;
      private readonly ACTIVE_INTERVAL = 1_000;
      private readonly IDLE_INTERVAL = 60_000;

      constructor(eventBus: EventBus) { ... }

      startPolling(plugId: string, ipAddress: string, intervalMs: number = 1000): void
        - Guard: if pollers.has(plugId), return (no double-polling)
        - Create async poll() function that:
          1. fetch(`http://${ipAddress}/rpc/Switch.GetStatus?id=0`) with 3s AbortSignal.timeout
          2. On success: build PowerReading { plugId, apower: status.apower ?? 0, voltage: status.voltage ?? 0, current: status.current ?? 0, output: status.output ?? false, totalEnergy: status.aenergy?.total ?? 0, timestamp: Date.now() }
          3. Emit via eventBus.emitPowerReading(reading)
          4. Call this.persistIfDue(reading)
          5. Update DB: plugs set online=true, lastSeen=Date.now()
          6. On failure (catch): update plugs set online=false, emit eventBus.emitPlugOnline(plugId, false)
        - Call poll() immediately, then setInterval(poll, intervalMs)
        - Store timer in pollers map

      stopPolling(plugId: string): void
        - Clear interval from pollers map, delete entry

      stopAll(): void
        - Clear all intervals, clear map

      isPolling(plugId: string): boolean
        - Return pollers.has(plugId)

      private persistIfDue(reading: PowerReading): void

        - If power > ACTIVE_POWER_THRESHOLD: persist every ACTIVE_INTERVAL (1s)
    ```

    ```typescript
    /**
     * Switch relay on via Shelly HTTP API.
     * Returns true if the command was acknowledged.
     */
    export async function switchRelayOnHttp(ipAddress: string): Promise<boolean> {
      try {
        const res = await fetch(
          `http://${ipAddress}/rpc/Switch.Set?id=0&on=true`,
          { signal: AbortSignal.timeout(3000) }
        );
        return res.ok;
      } catch {
        return false;
      }
    }

    /**
     * Switch relay off via Shelly HTTP API.
     * Returns true if the command was acknowledged.
     */
    export async function switchRelayOffHttp(ipAddress: string): Promise<boolean> {
      try {
        const res = await fetch(
          `http://${ipAddress}/rpc/Switch.Set?id=0&on=false`,
          { signal: AbortSignal.timeout(3000) }
        );
        return res.ok;
      } catch {
        return false;
      }
    }
    ```
