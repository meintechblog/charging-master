  Full implementation contract:

  ```ts
  import { spawn } from 'node:child_process';
  import { UpdateStateStore } from '@/modules/self-update/update-state-store';
  import type { UpdateTriggerResponse } from '@/modules/self-update/types';

  export const runtime = 'nodejs';
  export const dynamic = 'force-dynamic';

  const ALLOWED_HOSTS = new Set(['127.0.0.1', 'localhost', '::1', '[::1]']);
  const NO_CACHE = { 'Cache-Control': 'no-store, no-cache, must-revalidate' } as const;

  function isLocalhostHost(request: Request): boolean { /* same pattern as prepare-for-shutdown */ }

  export async function POST(request: Request): Promise<Response> {
    // 1) Host guard — 403 for non-localhost
    if (!isLocalhostHost(request)) {
      return Response.json({ status: 'error', error: 'forbidden' }, { status: 403, headers: NO_CACHE });
    }

    // 2) Parse optional body { targetSha?: string }
    let targetSha: string | null = null;
    try {
      const body = await request.json().catch(() => ({}));
      if (typeof body?.targetSha === 'string') targetSha = body.targetSha;
    } catch { /* ignore */ }

    // 3) Read state — must have updateAvailable true and not already running
    const store = new UpdateStateStore();
    let state;
    try { state = store.read(); }
    catch (err) {
      return Response.json({ status: 'error', error: `state read failed: ${err instanceof Error ? err.message : String(err)}` }, { status: 500, headers: NO_CACHE });
    }

    const updateAvailable = state.lastCheckResult?.status === 'ok' && state.lastCheckResult.remoteSha !== state.currentSha;
    if (!updateAvailable) {
      return Response.json({ status: 'error', error: 'no update available' }, { status: 409, headers: NO_CACHE });
    }
    if (state.updateStatus === 'installing') {
      return Response.json({ status: 'error', error: 'update already running' }, { status: 409, headers: NO_CACHE });
    }

    // Derive targetSha if not provided (fall back to remote from last check)
    if (targetSha === null && state.lastCheckResult?.status === 'ok') {
      targetSha = state.lastCheckResult.remoteSha;
    }
    if (targetSha === null) {
      return Response.json({ status: 'error', error: 'targetSha unknown' }, { status: 400, headers: NO_CACHE });
    }

    // 4) Pre-mark state (so a concurrent retry sees 'installing' and bounces)
    const startedAt = Date.now();
    try {
      store.write({ updateStatus: 'installing', targetSha, updateStartedAt: startedAt });
    } catch (err) {
      return Response.json({ status: 'error', error: `state write failed: ${err instanceof Error ? err.message : String(err)}` }, { status: 500, headers: NO_CACHE });
    }

    // 5) Fire-and-forget systemctl start. Every flag is load-bearing:
    //    - 'start --no-block' returns immediately (mitigates P1 parent-cgroup kill)
    //    - detached: true spawns into its own process group (extra safety)
    //    - stdio: 'ignore' closes stdin/out/err so the parent can exit cleanly
    //    - .unref() lets the Node event loop exit without waiting for the child
    //
    // On dev machine (macOS, no systemctl): spawn will error with ENOENT, OR
    // systemctl will exit code 4 (unit not found) / exit code 5 (unit not loaded).
    // In both cases, roll back the state write and return 503 with a dev-mode marker.
    try {
      const child = spawn(
        'systemctl',
        ['start', '--no-block', 'charging-master-updater.service'],
        { detached: true, stdio: 'ignore' }
      );
      child.unref();

      // Detect ENOENT synchronously via the 'error' event. We CAN'T await the
      // child (that would defeat --no-block), so we listen for a synchronous
      // error event that fires in the same tick as spawn() on ENOENT.
      // Pattern: wrap in a 200ms race so we catch ENOENT but still return fast.
      const spawnError = await new Promise<Error | null>((resolve) => {
        let resolved = false;
        const done = (e: Error | null) => { if (!resolved) { resolved = true; resolve(e); } };
        child.once('error', (e) => done(e));
        // Also catch exit=4/5 in the same tick for the case where systemctl
        // DOES exist (Linux dev) but the unit does not.
        child.once('exit', (code) => { if (code === 4 || code === 5) done(new Error(`systemctl exit ${code}`)); else done(null); });
        setTimeout(() => done(null), 200);
      });

      if (spawnError !== null) {
        // Roll back the state write so the UI doesn't get stuck in 'installing'.
        store.write({ updateStatus: state.updateStatus, targetSha: null, updateStartedAt: null });
        const isEnoent = (spawnError as NodeJS.ErrnoException).code === 'ENOENT' || /exit 4|exit 5/.test(spawnError.message);
        if (isEnoent) {
          return Response.json(
            { status: 'error', error: 'dev_mode: updater service not available on this host' },
            { status: 503, headers: NO_CACHE }
          );
        }
        return Response.json(
          { status: 'error', error: `spawn failed: ${spawnError.message}` },
          { status: 500, headers: NO_CACHE }
        );
      }
    } catch (err) {
      store.write({ updateStatus: state.updateStatus, targetSha: null, updateStartedAt: null });
      return Response.json(
        { status: 'error', error: `spawn threw: ${err instanceof Error ? err.message : String(err)}` },
        { status: 500, headers: NO_CACHE }
      );
    }

    const body: UpdateTriggerResponse = { status: 'triggered', startedAt, targetSha };
    return Response.json(body, { status: 202, headers: NO_CACHE });
  }
  ```

  Implementation notes for the executor:
  - The 200ms race is a pragmatic compromise: systemctl ENOENT errors fire synchronously on spawn, so 200ms is WAY more than enough to catch them. A real systemd start (Linux production) returns from `--no-block` in <50ms, so the 200ms window never actually delays production.
  - Do NOT use `execa` or any higher-level library — the brief calls out native `child_process.spawn` specifically.
  - Do NOT use `await` on the child process (would defeat --no-block).
  - Do NOT forget `.unref()` — without it the Node event loop can't exit cleanly if the child outlives the parent.
  - Host guard MUST be copy-pasted from prepare-for-shutdown/route.ts verbatim — do not invent a new guard.

  Implementation contract:

  ```ts
  import { spawn, type ChildProcess } from 'node:child_process';

  export const runtime = 'nodejs';
  export const dynamic = 'force-dynamic';

  const SSE_HEADERS = {
    'Content-Type': 'text/event-stream; charset=utf-8',
    'Cache-Control': 'no-cache, no-transform',
    'Connection': 'keep-alive',
    'X-Accel-Buffering': 'no', // disable nginx buffering even though we don't use nginx
  } as const;

  const ALLOWED_HOSTS = new Set(['127.0.0.1', 'localhost', '::1', '[::1]']);
  function isLocalhostHost(request: Request): boolean { /* same pattern */ }

  // SSE framing helper — wraps a raw log line as a data: event
  function frame(text: string): string {
    // Split on \n inside the line (journalctl should give us line-by-line but
    // be defensive) and emit each as its own data: entry, terminated by \n\n.
    const lines = text.split(/\r?\n/).filter(l => l.length > 0);
    return lines.map(l => `data: ${l}\n\n`).join('');
  }

  export async function GET(request: Request): Promise<Response> {
    if (!isLocalhostHost(request)) {
      return new Response('forbidden', { status: 403 });
    }

    const encoder = new TextEncoder();
    let child: ChildProcess | null = null;
    let heartbeatTimer: ReturnType<typeof setInterval> | null = null;
    let stdoutBuffer = '';

    // CRITICAL: cleanup() must be idempotent — called from BOTH
    // request.signal.abort AND ReadableStream cancel(). If journalctl is still
    // alive, SIGTERM it; if it survives 1s, SIGKILL.
    const cleanup = (): void => {
      if (heartbeatTimer !== null) {
        clearInterval(heartbeatTimer);
        heartbeatTimer = null;
      }
      if (child !== null && !child.killed) {
        const c = child;
        try { c.kill('SIGTERM'); } catch { /* ignore */ }
        setTimeout(() => {
          try { if (!c.killed) c.kill('SIGKILL'); } catch { /* ignore */ }
        }, 1000).unref();
        child = null;
      }
    };

    request.signal.addEventListener('abort', cleanup);

    const stream = new ReadableStream<Uint8Array>({
      start(controller) {
        // Attempt to spawn journalctl. If it fails (ENOENT on macOS dev, or
        // unit not found) we fall back to a synthetic stream.
        let spawned = false;
        try {
          child = spawn(
            'journalctl',
            ['-fu', 'charging-master-updater', '--output=cat', '--lines=100', '--no-pager'],
            { stdio: ['ignore', 'pipe', 'pipe'] }
          );
          spawned = true;

          child.on('error', (err) => {
            // ENOENT arrives here on macOS dev
            if ((err as NodeJS.ErrnoException).code === 'ENOENT') {
              controller.enqueue(encoder.encode(frame('[dev-mode] journalctl not available — synthetic stream')));
              controller.enqueue(encoder.encode(frame('[dev-mode] [stage=preflight] This is a dev-mode synthetic log.')));
              controller.enqueue(encoder.encode(frame('[dev-mode] Trigger the updater on a real LXC host to see real logs.')));
              return;
            }
            controller.enqueue(encoder.encode(frame(`[error] journalctl: ${err.message}`)));
          });

          child.stdout?.on('data', (chunk: Buffer) => {
            stdoutBuffer += chunk.toString('utf8');
            const lastNewline = stdoutBuffer.lastIndexOf('\n');
            if (lastNewline === -1) return;
            const complete = stdoutBuffer.slice(0, lastNewline);
            stdoutBuffer = stdoutBuffer.slice(lastNewline + 1);
            try { controller.enqueue(encoder.encode(frame(complete))); } catch { /* closed */ }
          });

          child.stderr?.on('data', (chunk: Buffer) => {
            try { controller.enqueue(encoder.encode(frame(`[stderr] ${chunk.toString('utf8').trim()}`))); } catch { /* closed */ }
          });

          child.on('exit', (code) => {
            // exit 4/5 = unit not known → fall back to synthetic dev stream
            if (code === 4 || code === 5) {
              try {
                controller.enqueue(encoder.encode(frame('[dev-mode] updater unit not installed — synthetic stream')));
                controller.enqueue(encoder.encode(frame('[dev-mode] [stage=preflight] synthetic')));
              } catch { /* closed */ }
            } else if (code !== null && code !== 0) {
              try { controller.enqueue(encoder.encode(frame(`[journalctl exited code=${code}]`))); } catch { /* closed */ }
            }
          });
        } catch (err) {
          // Synchronous throw (very rare) — synthetic fallback
          controller.enqueue(encoder.encode(frame(`[dev-mode] spawn threw: ${err instanceof Error ? err.message : String(err)}`)));
          controller.enqueue(encoder.encode(frame('[dev-mode] [stage=preflight] synthetic')));
        }

        // 10s heartbeat to keep the connection open and let the client notice
        // TCP death quickly. Emitted as an SSE comment (leading ':') so it
        // doesn't trigger the client's message handler.
        heartbeatTimer = setInterval(() => {
          try { controller.enqueue(encoder.encode(`: heartbeat ${Date.now()}\n\n`)); } catch { /* closed */ }
        }, 10_000);

        if (!spawned) {
          // Keep the stream open with just heartbeats — dev mode has no source
        }
      },
      cancel() {
        // Called when the client disconnects OR when Response is garbage-collected.
        // MUST cleanup even though request.signal.abort should also fire —
        // the brief explicitly says "BOTH hooks required" to prevent zombies.
        cleanup();
      }
    });

    return new Response(stream, { headers: SSE_HEADERS });
  }
  ```

  Critical details the executor must not miss:
  - Both `request.signal.addEventListener('abort', cleanup)` AND the ReadableStream `cancel()` callback MUST call `cleanup()`. The CONTEXT.md calls this out as load-bearing. In practice only one fires per disconnect but browsers differ — belt and braces.
  - The cleanup function is idempotent (guards on `child !== null && !child.killed`).
  - SIGTERM first, then SIGKILL after 1s. `setTimeout(...).unref()` so the timer doesn't keep the Node event loop alive.
  - Heartbeats as SSE comments (`: heartbeat\n\n`) so they're invisible to `onmessage` but keep the TCP connection alive.
  - Line-buffer `child.stdout` — journalctl can emit partial lines, we MUST only flush up to the last newline.
  - Catch ENOENT in the child's 'error' handler (macOS dev has no journalctl) and emit a synthetic dev stream instead of erroring.
  - Catch exit code 4 or 5 (unit not found on Linux dev) and fall back to the same synthetic stream.

  Security: Host-header guard only, same as trigger. Do NOT add per-user auth — there isn't one.

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → Next.js API routes | User clicks Install in the browser; the request traverses localhost HTTP and hits the route handlers. Browser JS is trusted (single-user LAN app) but the Host header is the only gate. |
| Node → systemd via `spawn(systemctl, ...)` | The trigger endpoint crosses from user-space Node into PID 1's job manager. systemctl inherits the service's privileges (root per ARCHITECTURE.md), which is why the Host guard must be airtight. |
| Node → journalctl child process | SSE log endpoint spawns journalctl and pipes its stdout to the browser. The child can emit arbitrary bytes; we frame them as SSE data but do not sanitize them (no HTML/JSON risk — the browser handles them as opaque log text). |
| UpdateStateStore ↔ Phase 9 bash script | state.json is a shared file between Node and shell. The shell writes rollbackStage/rollbackHappened during Phase 9's rollback trap; Node reads them via `read()`. Concurrent access is serialized by the fact that the updater runs in a oneshot unit that is a sibling of the main service — while the updater runs, the Node app has been stopped. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-10-01 | Spoofing | `/api/update/trigger` Host header | mitigate | `isLocalhostHost(request)` rejects anything not in `{127.0.0.1, localhost, ::1, [::1]}`. Copy-paste pattern from `src/app/api/internal/prepare-for-shutdown/route.ts` verbatim — any deviation from the existing audited guard is a bug. |
| T-10-02 | Tampering | State file write during concurrent trigger attempts | mitigate | Pre-mark `updateStatus='installing'` BEFORE spawn. A second concurrent POST reads the fresh state, sees `installing`, and returns 409. UpdateStateStore.write() is atomic (tmp + rename) so the read/write window cannot observe a partial file. |
| T-10-03 | Tampering | State file write from /api/update/ack-rollback with an unrelated flag | accept | The endpoint writes ONLY the three rollback fields. UpdateStateStore.write() merges via object spread — other fields are untouched. A hostile caller bypassing the Host guard (impossible without LAN access) could clear the rollback notice without running a real update, which would just hide a prior failure — low severity. |
| T-10-04 | Repudiation | Updater launched without a trace | mitigate | Every trigger writes `updateStartedAt` + `targetSha` to state.json before spawn; Phase 9 already logs to `update_runs` table. A dev-mode 503 rollback-write ensures a failed trigger is not silently recorded as 'installing'. |
| T-10-05 | Information Disclosure | SSE log leaks secrets via journalctl | accept | Rationale: `journalctl -fu charging-master-updater` only contains the updater unit's stdout/stderr. Pushover token and GitHub PAT live in the `config` table and are read by the main service (different unit) — they are NOT in the updater's journal. Verified by grepping scripts/update/run-update.sh — no secret values pass through stdout. |
| T-10-06 | Denial of Service | SSE endpoint spawns unbounded journalctl children | mitigate | One journalctl child per open SSE connection. Cleanup on BOTH request.signal.abort AND ReadableStream cancel() — any disconnect kills the child. SIGTERM + 1s SIGKILL fallback for stubborn children. Single-user app — the only way to exhaust the process table is for the user to open 10,000 tabs, which is not in the threat model. |
| T-10-07 | Denial of Service | Concurrent trigger fills job queue | mitigate | `updateStatus === 'installing'` check returns 409 for the second caller. flock in the updater unit (Phase 9) is a second line of defense. |
| T-10-08 | Denial of Service | Zombie journalctl process survives disconnect | mitigate | Double cleanup hook (signal.abort + ReadableStream.cancel). Verified in Task 3 by `ps -ax | grep journalctl` returning empty after curl closes. |
| T-10-09 | Elevation of Privilege | API route gains ability to run arbitrary shell via spawn | mitigate | `spawn(cmd, argv, opts)` is called with a literal command string and a fixed argv array — NO shell interpolation, NO user input passed to the command line. The only user-controlled input (targetSha from the POST body) is written to state.json via UpdateStateStore.write(), not passed to systemctl. |
| T-10-10 | Elevation of Privilege | Detached child escapes cgroup kill on main service restart | mitigate | `spawn(..., { detached: true, stdio: 'ignore' }).unref()` combined with `systemctl start --no-block` ensures the updater runs under its own cgroup (owned by systemd PID 1), not as a child of charging-master.service.
